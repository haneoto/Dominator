# Mihomo Party DNS设置备份
# 大陆DNS分流参考：https://github.com/SukkaW/Surge/blob/master/Source/non_ip/domestic.ts
# 境外代理域名走远程解析，无需DNS分流。

# nameserverPolicy对应应用内DNS模块的“覆盖DNS策略”，通过配置文件编辑时需先在\data\config.yaml中修改。
# nameserverPolicy覆写优先级：config.yaml > mihomo.yaml > 覆写模块
# config.yaml修改完毕后重启应用并重启DNS模块开关，Mihomo Party会将该字段重新写入到内核配置（\data\mihomo.yaml）与当前配置（\data\work\config.yaml）中。在应用内修改相关设置会直接写入到这两个配置文件中。

nameserverPolicy:
  geosite:private:
    - system
  geosite:category-ads-all: rcode://success
  geosite:alibaba,aliyun,youku:
    - quic://223.5.5.5:853
    - quic://223.6.6.6:853
  geosite:tencent,qcloud,bilibili@cn,xiaomi:
    - https://1.12.12.12/dns-query
    - https://120.53.53.53/dns-query
  geosite:bytedance:
    - 180.184.1.1
    - 180.184.2.2
  geosite:baidu,iqiyi:
    - 180.76.76.76
  geosite:cn:
    - quic://223.5.5.5:853
    - https://1.12.12.12/dns-query


# nameserver、proxy-server-nameserver、hosts分别对应DNS模块的”DNS 服务器“、”代理节点域名解析“、”自定义 Hosts“，通过配置文件编辑时在\data\mihomo.yaml中修改，重启应用后变更会反应在界面上。
    
  nameserver:
    - quic://223.5.5.5
    - https://1.12.12.12/dns-query
  proxy-server-nameserver:
    - quic://223.5.5.5
    - https://1.12.12.12/dns-query
